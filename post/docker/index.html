nginx

第一次运行

docker run -d --name nginx -p 80:80 nginx

新建挂载目录

mkdir -p /root/docker/nginx/{conf, logs, html, ssl, conf/conf.d}

拷贝文件信息

docker cp nginx:/etc/nginx/nginx.conf /root/docker/nginx/conf && docker cp nginx:/etc/nginx/conf.d/default.conf /root/docker/nginx/conf/conf.d/default.conf

配置index.html

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>hello</title>
</head>
<body>
    <h1>
        hello, world !
    </h1>
</body>
</html>

停止容器并重新启动容器

docker container stop nginx && docker container rm nginx

docker run \
 -d \
 --rm \
 -p 80:80  \
 -p 443:443  \
 --name nginx \
 -v /root/docker/nginx/html:/usr/share/nginx/html \
 -v /root/docker/nginx/conf/nginx.conf:/etc/nginx/nginx.conf \
 -v /root/docker/nginx/conf/conf.d/default.conf:/etc/nginx/conf.d/default.conf \
 -v /root/docker/nginx/logs:/var/log/nginx \
 -v /root/docker/nginx/ssl:/ssl \
 nginx

redis

拉取Redis镜像

docker pull redis

第一次运行

docker run -d --name redis -p 6379:6379 redis

新建挂载目录

mkdir -p /root/docker/redis/{conf, data}

获取redis - conf

sudo wget https://raw.githubusercontent.com/antirez/redis/5.0/redis.conf -O /root/docker/redis/conf/redis.conf

修改conf
# Redis默认不是以守护进程的方式运行，可以通过该配置项修改，使用yes启用守护进程
protected-mode no
# 开启远程
deamonize  no
# #注释掉这部分，这是限制redis只能本地访问,你可以绑定单一接口，如果没有绑定，所有接口都会监听到来的连接
#  bind 127.0.0.1     
# 因为redis本身同步数据文件是按上面save条件来同步的，所以有的数据会在一段时间内只存在于内存中。默认为no
appendonly yes
# 设置Redis连接密码，如果配置了连接密码，客户端在连接Redis时需要通过
# auth <password>命令提供密码，默认关闭，当前密码为123456
requirepass password

删除原容器

docker container stop redis && docker container rm redis

启动Redis

docker run -p 6379:6379 \
--name redis \
--restart=always \
--privileged=true \
-v /root/docker/redis/data:/data \
-v /root/docker/redis/conf/redis.conf:/etc/redis/redis.conf \
-d redis redis-server /etc/redis/redis.conf \
--appendonly yes \
--requirepass 'password'


docker run -p 6379:6379 \
--name redis \
--privileged=true \
-v /root/docker/redis/data:/data \
-v /root/docker/redis/conf/redis.conf:/etc/redis/redis.conf \
-d redis redis-server /etc/redis/redis.conf


docker run -d \
--privileged=true \
-p 6379:6379 \
--restart always \
-v /usr/local/redis/redis.conf:/etc/redis/redis.conf \
-v /usr/local/redis/data:/data \
--name myredis redis:6.0 redis-server /etc/redis/redis.conf \
--appendonly yes \
--requirepass "123456"

命令解析

  • –restart=always 是随Docker启动
  • redis-server –appendonly yes 是启动时执行命令设置持久化
  • –requirepass admin@798 是设置连接密码为: admin@798
