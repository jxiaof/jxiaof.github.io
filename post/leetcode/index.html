class Solution:
    def maxProfit(self, prices):
        """
        :type prices: List[int]
        :rtype: int
        """
        min_p, max_p = 999999, 0
        for i in range(len(prices)):
            min_p = min(min_p, prices[i])
            max_p = max(max_p, prices[i] - min_p)
        return max_p

from collections import OrderedDict

class LRUCache:

    def __init__(self, size):

        self.size = size

        self.linked_map = OrderedDict()

    def set(self, key, value):

        if key in self.linked_map:

            self.linked_map.pop(key)

        if self.size == len(self.linked_map):

            self.linked_map.popitem(last=False)

        self.linked_map.update({key: value})

    def get(self, key):

        value = self.linked_map.get(key)

        self.linked_map.pop(key)

        self.linked_map.update({key: value})

        return value

def BinarySearch(array: list, t: int) -> int:
    '''
    index a item in object array
    :param array: [1, 2, 3, 34, 56, 57, 78, 90, 107]
    :param t: 4
    :return: item index
    '''
    low = 0
    height = len(array) - 1
    while low < height:
        mid = (low+height) // 2
        if array[mid] < t:
            low = mid + 1
        elif array[mid] > t:
            height = mid - 1
        else:
            return mid
    return height if array[low] == t else f"Not exist in object array, this new value may be place in: {height} !"

最长节点

给定一个二叉树，找出其最大深度。

二叉树的深度为根节点到最远叶子节点的最长路径上的节点数。

说明: 叶子节点是指没有子节点的节点。

示例： 给定二叉树 [3,9,20,null,null,15,7]，

    3
   / \
  9  20
    /  \
   15   7

返回它的最大深度 3 。

class Solution:
    def maxDepth(self, root: TreeNode) -> int:
        if root == None:
            return 0
        elif root.right == None and root.left == None :
            return 1
        else:
            return 1 + max(self.maxDepth(root.left) ,  self.maxDepth(root.right))

class Solution(object):
    def maximumGap(self, nums):
        """
        [3,6,9,1]
        :type nums: List[int]
        :rtype: int
        """
        max_num, flag = 0, False
        for i in range(len(nums)):
            min_idx = i
            for j in range(i + 1, len(nums)):
                if nums[min_idx] > nums[j]:
                    min_idx = j
            nums[i], nums[min_idx] = nums[min_idx], nums[i]
            if flag is True:
                if max_num < nums[i] - nums[i-1]:
                    max_num = nums[i] - nums[i-1]
            else:
                flag = True
        return nums, max_num

def synchronized(func):

    func.__lock__ = threading.Lock()

    def lock_func(*args, **kwargs):
        with func.__lock__:
            return func(*args, **kwargs)
    return lock_func

class CallLimitor(object):
    """
    函数访问限制器
    """
    # limit_time = 60
    # limit_times = 2
    # cache = {}

    def __init__(self, limit_time=60, limit_times=2):
        self.limit_time = limit_time
        self.limit_times = limit_times
        self.cache = {}
        self.lock = threading.Lock()

    def __call__(self, func, *args, **kwargs):
        # 函数的名称作为key
        key = func.__name__
        # self.func_name = key
        # 判断是否存在缓存
        # if key in self.cache.keys():
        with self.lock:
            if key in self.cache.keys():
                (counter, result, updateTime) = self.cache[key]
                # 过期时间固定为 limit_time 秒
                time_gap = time.time() - updateTime
                if time_gap < self.limit_time:
                    counter += 1
                # 超过限制时间，函数被释放，重置相关变量
                else:
                    print(f"cache expired ! func:【 {key} 】can call !")
                    result = None
                    counter = 1
                    updateTime = time.time()
            else:
                # 第一次调用函数，初始化相关变量
                print(f"no cache for func:【 {key} 】!")
                result = None
                counter = 1
                updateTime = time.time()
            # 如果过期，或则没有缓存调用方法
            if counter <= self.limit_times:
                result = func(*args, **kwargs)
                # with self.lock:
                self.cache[key] = (counter, result, updateTime)
                print(f"already called {counter} times !")
            else:
                print(
                    f"func name:【 {key} 】 call deny ! limit {self.limit_time}s call {self.limit_times} times, already call denied {time_gap: .3f}s !")

        return result


def call_limiter(limit_time, limit_times):

def call_limiter_lock(limit_time, limit_times):
    """
    限制函数访问次数,适用于单进程多线程环境
    :param limit_time: 访问限制时间
    :param limit_times: 访问限制时间内限制访问次数
    :return:
    """

    def outter(func):
        "cache for function result, which is immutable with fixed arguments"
        print(f"initial cache for {func.__name__}")
        cache = {}
        lock = threading.Lock()

        @wraps(func)
        def inner(*args, **kwargs):
            # 函数的名称作为key
            key = func.__name__
            # 判断是否存在缓存
            # if key in cache.keys():  # 先判断,然后再加锁,防止多线程同时调用函数频繁加锁
            with lock:
                # 判断是否存在缓存
                if key in cache.keys():
                    (counter, result, updateTime) = cache[key]
                    # 过期时间固定为 limit_time 秒
                    time_gap = time.time() - updateTime
                    if time_gap < limit_time:
                        counter += 1
                    # 超过限制时间，函数被释放，重置相关变量
                    else:
                        print(f"cache expired ! func:【 {key} 】can call !")
                        result = None
                        counter = 1
                        updateTime = time.time()
                else:
                    # 第一次调用函数，初始化相关变量
                    print(f"no cache for func:【 {key} 】!")
                    result = None
                    counter = 1
                    updateTime = time.time()
                # 如果过期，或则没有缓存调用方法
                if counter <= limit_times:
                    result = func(*args, **kwargs)
                    # with lock:
                    cache[key] = (counter, result, updateTime)
                    print(f"already called {counter} times !")
                else:
                    print(
                        f"func name:【 {key} 】 call deny ! limit {limit_time}s call {limit_times} times, already call denied {time_gap: .3f}s !")

                return result

        return inner

    return outter
